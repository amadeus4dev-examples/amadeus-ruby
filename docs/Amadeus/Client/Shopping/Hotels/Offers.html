def get(params = {})
  @offer_id ||= begin
    offer_id = params[:id] || params['id']
    params.delete(:id)
    params.delete('id')
    offer_id
  end

  client.get(
    "/v1/shopping/hotel/#{@hotel_id}/offers/#{@offer_id}", params
  )
end
