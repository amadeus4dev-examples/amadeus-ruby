def initialize(http_response, client, verb=nil, path=nil, params=nil)
  @http_response = http_response
  @client = client

  # used for pagination
  @verb = verb
  @path = path
  @params = params

  log_response
  parse_response
end

def log_response
  @client.logger.debug('Amadeus::Response') do
    # :nocov:
    "#{http_response.class.name}\n#{http_response.body}"
    # :nocov:
  end
end
